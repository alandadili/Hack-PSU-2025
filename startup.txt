App Startup:
- Backend:
    - python -m venv .venv
    - .\.venv\Scripts\Activate
    - pip install -r .\requirements.txt
    - python -m uvicorn main:app --reload --host 127.0.0.1 --port 8000
- Frontend:
    - npm install
    - npm start

Running:
(/backend) python -m uvicorn main:app --reload --host 127.0.0.1 --port 8000
(/frontend) npm start
